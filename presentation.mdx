import ProgressBarInAndOut from './src/components/Demo/ProgressBar.fade-in-and-out';

<Value initial={0.5}>
  {({ value, set }) => (
    <Toggle initial={true}>
      {({ on, toggle }) => (
        <React.Fragment>
          <ProgressBarInAndOut
            width={300}
            height={15}
            progress={value}
            isVisible={on}
